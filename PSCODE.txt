<!--- SOAL NOMOR 2 TAHUN KABISAT --->

Start Algoritma
1. Tetapkan nilai Tahun = "" ;
2. Periksa nilai Tahun Kabisat dengan kondisi Jika Tahun habis dibagi 4 dan tidak habis dibagi 100 dan Jika Tahun habis
   dibagi 4 habis di bagi 100 dan habis dibagi 400;
  2a. Jika Nilai Tahun kabisat maka akan muncul "BENAR INI TAHUN KABISAT";
  2b. Jika Nilai Tahun Bukan Kabisat maka akan muncul "INI BUKAN TAHUN KABISAT";

END Algoritma

-----------------------------------------------------------------------------------------------------

Start Pseudocode

STORE "tahun" to any number;
IF "tahun" mod 4 === 0 && "tahun" mod 100 !== 0
    DISPLAY "ini tahun kabisat"
ELSE IF "tahun" mod 4 && "tahun" mod 100 === 0 && "tahun" mod 400
    DISPLAY "ini tahun kabisat"
Else
    DISPLAY " ini bukan tahun kabisat"

ENDIF

end Pseudocode



<!--- SOAL NOMOR 3 LAUNDRY DAY --->

1. Mesin cuci di set dengan angka 0;
2. Setiap kali Foxie memasukan pakaian kedalam mesin cuci, maka nilai mesin cuci akan bertambah +1;
3. kondisi ini akan mengecek apakah jumlah pakai di mesin cuci sudah 20 jika belum akan terus berulang sampai jumlah pakaian di mesin cuci berjumlah 20;
4. setelah berjumlah 20 mesin cuci bekerja;

End Algoritma

-----------------------------------------------------------------------------------------------------

Start Pseudocode

FOR STORE "jumlahCuci" WITH 0; "jumlahCuci" < 20; "jumlahCuci" equals "jumlahCuci" plus 1
  IF "jumlahCuci" LESS 20
    DISPLAY "Masukan Lagi pakaiannya ke dalam mesin cuci"
  ELSE IF "jumlahCuci" equals 20
    DISPLAY "Mesin Cuci Siap dijalankan"
  ElSE
    DISPLAY "BELUM DIMASUKAN SAMA SEKALi"


end Pseudocode
